\[ \sin^2 B - \sin^2 A = \frac{1}{2}\sin^2 C \]

\[ \sin^2 B - \frac{1}{4} = \frac{1}{2}\sin^2 C \]

\[ \sin B = \sin (A + C) = \sin A \cos C + \cos A \sin C \]

\[ \sin B = \frac{1}{2} \cos C + \frac{\sqrt{3}}{2} \sin C \]
